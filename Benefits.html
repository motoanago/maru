function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getActiveSheet(); const lastRow = sheet.getLastRow(); let data = []; for (let r = 2; r <= lastRow; r++) { // A列：公開日（文字列として取得） let published = sheet.getRange(r, 1).getDisplayValue(); // B列：タイトル let title = sheet.getRange(r, 2).getDisplayValue(); // C列：リンク（スマートチップ対応） let urlCell = sheet.getRange(r, 3); let formula = urlCell.getFormula(); let url = ""; if (formula && formula.startsWith("=HYPERLINK(")) { // =HYPERLINK("URL","名前") 形式からURLを抽出 const match = formula.match(/"(.*?)"/); if (match) url = match[1]; } else { url = urlCell.getDisplayValue(); // 通常のURL } // D列：接続パスワード let password = sheet.getRange(r, 4).getDisplayValue(); if (title && url) { data.push([published, title, url, password]); } } let html = ` リンク集

リンク集

`; data.forEach(row => { let [published, title, url, password] = row; html += `
${published}
${title}
開く ${password ? `
パスワード: ${password}
` : ""}
`; }); html += ` `; return HtmlService.createHtmlOutput(html) .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); }
